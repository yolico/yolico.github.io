<template>
<d2-container>
    <div class="content">
    <!-- 基本信息区域 -->
    <div class="title">答题</div>
    <div class="little-title" v-if="showSub < Slength + 1">文字单选题( {{showSub}}/{{Slength}} ｜ 共{{Slength}}题 )<span>用时：00:00:{{time1}}</span></div>
    <div class="little-title" v-else-if="showSub - Slength < Mlength + 1">文字多选题( {{showSub - Slength}}/{{Mlength}} ｜ 共{{Mlength}}题 )<span>用时：00:00:{{time1}}</span></div>
    <div class="little-title" v-else-if="showSub - Slength -  Mlength< Rlength + 1">判断题( {{showSub - Slength - Mlength}}/{{Rlength}} ｜ 共{{Rlength}}题 )<span>用时：00:00:{{time1}}</span></div>
    <!-- 题目区域 -->
    <div class="main">
    <!-- 题目区域 -->
        <div class="subject">
        <!-- 单选题 -->
        <ul v-if="showSub < Slength + 1">
            <li v-for="(sitem , index) in single" :key="index" v-if="showSub === index + 1">
            <div class="subject-title">
                {{index + 1}} , {{sitem.content}} ( )
            </div>
            <el-radio-group v-model="sitem.own" class="subject-select">
                <el-radio :label="s.option" v-for="(s ,index) in sitem.options" :key="index">{{s.option}}. {{s.content}}</el-radio>
            </el-radio-group>
            </li>
        </ul>
        <!-- 多选题 -->
        <ul v-else-if="showSub - Slength < Mlength + 1">
            <li v-for="(mitem , index) in multiple" :key="index" v-if="showSub - Slength === index + 1">
            <div class="subject-title">
                {{index + 1}} , {{mitem.content}} ( )
            </div>
            <!-- <el-radio-group v-model="radio" class="subject-select">
                <el-radio v-model="radio" :label="m.id" v-for="(m ,index) in mitem.optionLists" :key="index">{{m.option}}. {{m.content}}</el-radio>
            </el-radio-group> -->
            <el-checkbox-group v-model="mitem.own" class="subject-select" @change="Msbtn(mitem.id)">
                <el-checkbox :label="m.option" v-for="(m ,index) in mitem.options" :key="index">{{m.option}}. {{m.content}}</el-checkbox>
            </el-checkbox-group>
            </li>
        </ul>
        <!-- 判断题 -->
        <ul v-else-if="showSub - Slength -  Mlength < Rlength + 1">
            <li v-for="(ritem , index) in recognized" :key="index" v-if="showSub - Slength -  Mlength === index + 1">
            <div class="subject-title">
                {{index + 1}} , {{ritem.content}} ( )
            </div>
            <el-radio-group v-model="ritem.own" class="subject-select" @change="Rsbtn(ritem.id)">
                <el-radio label="0">错误</el-radio>
                <el-radio label="1">正确</el-radio>
            </el-radio-group>
            </li>
        </ul>
        <div class="btn">
            <el-button class="left" type="primary" @click="lastTime" v-if="showSub > 1">上一题</el-button>
            <el-button class="right" type="primary" @click="anotherTime" v-if="showSub - Slength - Mlength - Rlength < 0">下一题</el-button>
        </div>
        </div>
        <!-- 答题卡区域 -->
        <div class="answerSheet">
        <div class="singleChoiceBox">
            <div class="answerTitle">单选题</div>
            <ul>
            <!-- :style="{ 'backgroundColor':single[item].own  ? 'blue':''}" -->
            <li class="itm" v-for="item in Slength" :key="item" @click="sBtn(item)" :style="{ 'backgroundColor' : single[item - 1].own  ? 'skyblue':''}">{{item}}</li>
            </ul>
        </div>
        <div class="multipleChoiceBox">
            <div class="answerTitle">多选题</div>
            <ul>
            <li class="itm" v-for="item in Mlength" :key="item" @click="mBtn(item)" :style="{ 'backgroundColor' : multiple[item - 1].own.length  ? 'skyblue':''}">{{item}}</li>
            </ul>
        </div>
        <div class="recognizedChoiceBox">
            <div class="answerTitle">判断题</div>
            <ul>
            <li class="itm" v-for="item in Rlength" :key="item" @click="rBtn(item)" :style="{ 'backgroundColor' : recognized[item - 1].own  ? 'skyblue':''}">{{item}}</li>
            </ul>
        </div>
        <!-- 交卷 -->
        <div class="handin">
            <el-button type="primary" @click="handin">交卷</el-button>
            </div>
        </div>
        </div>
    </div>
</d2-container>
</template>

js部分

export default {
        name: 'test',
        data () {
            return {
            // 题目显示
            showSub: 1,
            // 多选题
            multiple: [],
            Mlength: '',
            // 单选题
            single: [],
            Slength: '',
            // 判断题
            recognized: [],
            Rlength: '',
            // 提交的答案
            lists: []
            }
        },
        computed: {
            ...mapState('d2admin', {
            // 考试订单id
            exId: state => state.exam.exId,
            // 考试课程id
            coId: state => state.exam.coId
            })
        },
        created () {
            // this.countDown()
            this.getTM()
        },
        methods: {
            ...mapMutations('d2admin/exam', [
            'savePostId'
            ]),
            // 倒计时
            countDown () {
            let that = this
            this.timer = setInterval(() => {
                that.time1--
                if (that.time1 < 1) {
                // alert('您的答题时间已到,现在为您进入下一题')
                clearInterval(that.timer)
                }
            }, 1000)
            },
            // 交卷
            handin () {
            // 处理单选题
            this.single.map(stm => {
                let sobj = {}
                sobj.answer = stm.own
                sobj.id = stm.id
                this.lists.push(sobj)
            })
            // 处理多选题
            this.multiple.map(mtm => {
                let mobj = {}
                mobj.answer = mtm.own.join()
                mobj.id = mtm.id
                this.lists.push(mobj)
            })
            // 处理判断题
            this.recognized.map(rtm => {
                let robj = {}
                robj.answer = rtm.own
                robj.id = rtm.id
                this.lists.push(robj)
            })
            sendTest({ courseInfoId: this.coId, lists: this.lists, orderId: this.exId }).then(res => {
                if (res.code === 0) {
                this.savePostId(res.data)
                this.$emit('toHandin')
                }
            }).catch(() => {
                return false
            })
            },
            // 初始化题目(主要在于处理后端返回的数据格式,将其处理成适合渲染题目格式的数据结构)
            getTM () {
            getTest(this.coId).then(res => {
                if (res.code === 0) {
                // 处理多选
                this.Mlength = res.data.multipleChoice.length
                res.data.multipleChoice.map((item, index) => {
                    let obj = {}
                    // 题目id
                    obj.id = item.id
                    // 题目内容
                    obj.content = item.content
                    // 选项
                    obj.options = item.optionLists
                    // 选中的答案
                    obj.own = []
                    // 题目类型
                    obj.type = 'm'
                    this.multiple.push(obj)
                })
                // 处理单选
                this.Slength = res.data.singleChoice.length
                res.data.singleChoice.map((item, index) => {
                    let obj = {}
                    // 题目id
                    obj.id = item.id
                    // 题目内容
                    obj.content = item.content
                    // 选项
                    obj.options = item.optionLists
                    // 选中的答案
                    obj.own = ''
                    // 题目类型
                    obj.type = 's'
                    this.single.push(obj)
                })
                // 处理判断
                this.Rlength = res.data.recognizedChoice.length
                res.data.recognizedChoice.map((item, index) => {
                    let obj = {}
                    // 题目id
                    obj.id = item.id
                    // 题目内容
                    obj.content = item.content
                    // 选项
                    obj.options = item.optionLists
                    // 选中的答案
                    obj.own = ''
                    // 题目类型
                    obj.type = 'r'
                    this.recognized.push(obj)
                })
                }
            }).catch(() => {
                return false
            })
            },
            // 单选题答案
            Ssbtn (sum) {
            this.testId = sum
            },
            // 多选题答案
            Msbtn (mum) {
            this.testId = mum
            },
            // 判断题答案
            Rsbtn (rum) {
            this.testId = rum
            },
            // 上一题
            lastTime () {
            this.showSub--
            },
            // 下一题
            anotherTime () {
            this.showSub++
            },
            // 答题卡相关跳转
            sBtn (s) {
            this.showSub = s
            },
            mBtn (m) {
            this.showSub = m + this.Slength
            },
            rBtn (r) {
            this.showSub = r + this.Slength + this.Mlength
            }
            }
        }
